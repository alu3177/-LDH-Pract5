25 bool TuringTape::ParseFile(string const &i_path){
26  fstream* fs = new fstream(i_path.c_str());
27 
28  if (!fs->is_open()){
29  cout << "[ERROR FICHERO] No se ha podido abrir el fichero \"" << i_path << "\"" << endl;
30  return false;
31  }
32 
33  string raw;
34 
35  *fs >> raw;
36 
37  for (uint16_t i = 0; i < raw.size(); i++)
38  Write(i, raw[i]);
39  return true;
40 }

50 void TuringTape::Write(int const &i_pos, char const &i_symbol){
51  if (i_pos < _lowerPos)

53  else if (i_pos > _upperPos)
54  _upperPos = i_pos;
55  _tape[i_pos] = i_symbol;
