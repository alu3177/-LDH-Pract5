33 bool TuringMachine::ParseFile(string const &i_path){
34  fstream* fs = new fstream(i_path.c_str());
35  if (!fs->is_open()){
36  cout << "[ERROR FICHERO] No se ha podido abrir el fichero \"" << i_path << "\"" << endl;
37  return false;
38  }
39 
40  int nStates;
41  int firstFinal;
42  int nTuples;
43 
44  *fs >> nStates;
45  *fs >> firstFinal;
46  *fs >> nTuples;
47 
48  for (uint16_t i = 0; i < nTuples; i++){
49  Transition* tr = new Transition;
50  char move;
51  *fs >> tr->stateID >> tr->inputSymbol >> tr->writeSymbol >> move >> tr->nextStateID;
52  switch(move){
53  case 'R':
54  case 'r': tr->move = R; break;
55  case 'L':
56  case 'l': tr->move = L; break;
57  case 'S':
58  case 's': tr->move = S; break;
59  }
60 
61  State* stActual = GetState(tr->stateID);
62  if (stActual == NULL){ // No existe el estado
63  bool final = tr->stateID < firstFinal ? false : true;
64  stActual = new State(tr->stateID, final);
65  AddState(stActual);
66 
67  }
68  stActual->AddTransition(tr);
69  }
70  // Manejo de estados sin transiciones
71  if (_states->size() < nStates)
72  for (idt_t i = 0; i < nStates; i++)
73  if (GetState(i) == NULL){
74  AddState(new State(i, i < firstFinal ? false : true));
75  }
76  return true;
77 }

84 bool TuringMachine::Run(){
85  int nSteps = 0;
86  State* stActual = GetState(0);
87  _tapePos = stActual->GetID();
88  if (VERBOSE)
89  cout << nSteps++ << " - ";
90  cout << "[" << 0 << "] ";
91  for (int i = _tape->LowerPos() - 1; i <= _tape->UpperPos() + 1; i++){
92  if (i == 0)
93  cout << C_BRED;
94  cout << _tape->Read(i);
95  if (i == 0)
96  cout << C_DEFAULT;
97  }
98  cout << endl;
99 
100  while(stActual->TransitionCount() > 0){
101  char symbLeido = _tape->Read(_tapePos);
102  Transition* trActual = stActual->GetTransition(symbLeido);
103  if (trActual == NULL){
104  cout << C_BRED << "[RESULTADO] Estado de NO aceptación." << C_DEFAULT << endl; // DEBUG ?
105  return (stActual->isFinal() ? true : false);
106  }
107  _tape->Write(_tapePos, trActual->writeSymbol);
108  switch(trActual->move){
109  case L : _tapePos--; break;
110  case R : _tapePos++; break;
111  default : break;
112  }
113  stActual = GetState(trActual->nextStateID);
114  if (stActual == NULL){
115  cout << C_BRED << "[RESULTADO] Estado de NO aceptación." << C_DEFAULT << endl; // DEBUG ?
116  return false;
117  }
118  // Mostramos las cinta, resaltando en color la posición del cabezal
119  if (VERBOSE)
120  cout << nSteps++ << " - ";
121  cout << "[" << stActual->GetID() << "] ";
122  for (int i = _tape->LowerPos() - 1; i <= _tape->UpperPos() + 1; i++){
123  if (i == _tapePos)
124  cout << C_BRED;
125  cout << _tape->Read(i);
126  if (i == _tapePos)
127  cout << C_DEFAULT;
128  }
129  cout << endl;
130  }
131  if (stActual->isFinal()){
132  cout << C_BGREEN << "[RESULTADO] Estado de ACEPTACIÓN." << C_DEFAULT << endl;
133  return true;
134  }
135  return false;
136 }
